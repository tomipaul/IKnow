ALTER TABLE user_connections
ADD FOREIGN KEY(user_id) REFERENCES users (id),
ADD FOREIGN KEY(connection_id) REFERENCES connections (id);
ALTER TABLE resources
ADD FOREIGN KEY(user_id) REFERENCES users (id) ON DELETE CASCADE;
ALTER TABLE resource_tags
ADD FOREIGN KEY(resource_id) REFERENCES resources (id) ON DELETE CASCADE,
ADD FOREIGN KEY(tag_id) REFERENCES tags (id) ON DELETE CASCADE;
ALTER TABLE recommendations
ADD FOREIGN KEY(resource_id) REFERENCES resources (id) ON DELETE CASCADE,
ADD FOREIGN KEY(user_id) REFERENCES users (id) ON DELETE CASCADE;
ALTER TABLE comments
ADD FOREIGN KEY(user_id) REFERENCES users (id),
ADD FOREIGN KEY(resource_id) REFERENCES resources (id);
ALTER TABLE resource_collections
ADD FOREIGN KEY(resource_id) REFERENCES resources (id) ON DELETE CASCADE,
ADD FOREIGN KEY(collection_id) REFERENCES collections (id) ON DELETE CASCADE;
